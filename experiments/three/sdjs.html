Hello 3D World

                var camera,scene,cube_mesh;
                init();
                animate();

                // init
                function init(){
                    //camera
                    camera = new THREE.Camera( 75, window.innerWidth / window.innerHeight, 1, 10000 );
                    camera.position.z = 250;
                    camera.position.y = 100;
                    //cube
                    var cube_material = new THREE.MeshBasicMaterial( {color: 0xffe433 });
                    var cube_geometry = new THREE.CubeGeometry( 100, 100, 100 );
                    cube_mesh = new THREE.Mesh(cube_geometry,cube_material);
                    // lights
                    var light = new THREE.PointLight(0xffffff);
                    light.position.y = 500;
                    //scene
                    scene = new THREE.Scene();
                    scene.addLight(light);
                    scene.addObject(cube_mesh);
                    // renderer
                    renderer = new THREE.CanvasRenderer();
                    renderer.setSize( window.innerWidth, window.innerHeight );
                    document.body.appendChild( renderer.domElement );
                }
                // animate loop
                function animate() {
                    requestAnimationFrame( animate );
                    renderer.render( scene, camera );
                    cube_mesh.rotation.y += 0.01;
                }
